html to markdown 油猴脚本

Created2024-01-23|Updated2024-01-23|项目
|Word count:822|Reading time:3min|Post View:

目标

在手机上实现 html 转为 markdown 格式，并将其存入 obsidian 来进行长期消化。

脚本需求

  • 使用手动输入网页标签实现解析
  • 输入网页标签后自动保存在脚本中
  • 是一个悬浮窗样式，可以停靠在页面的边缘
  • 可以设置黑名单，让某些网页不显示悬浮窗
  • 当第一次访问一个网页时，让你输入网页 html 标签来进行解析，如果没有解析到，则重复输入，第二次访问相应的网页时，不会显示输入框，点击悬浮窗时会将解析后的文本复制在剪切板里

调用库

在解析 html 时，调用如下库：

  • turndown 库：进行基本的解析
  • turndown-plugin-gfm：进行代码块和表格的解析

源码

(function () {
  "use strict";

  // 创建悬浮球元素
  var floatingButton = document.createElement("div");
  floatingButton.innerHTML = "MD";
  floatingButton.style.cssText =
    "position: fixed; top: 50%; left: 0; z-index: 9999; background-color: red; color: #fff; width: 50px; height: 50px; border-radius: 50%; text-align: center; line-height: 50px; cursor: move;";
  document.body.appendChild(floatingButton);

  var isDragging = false;
  var startPositionX = 0;
  var startPositionY = 0;
  var offsetX = 0;
  var offsetY = 0;

  floatingButton.addEventListener("mousedown", function (event) {
    isDragging = true;
    startPositionX = event.clientX;
    startPositionY = event.clientY;
    offsetX = floatingButton.offsetLeft;
    offsetY = floatingButton.offsetTop;
  });

  document.addEventListener("mousemove", function (event) {
    if (isDragging) {
      var currentX = offsetX + event.clientX - startPositionX;
      var currentY = offsetY + event.clientY - startPositionY;

      var windowWidth = window.innerWidth;
      var windowHeight = window.innerHeight;
      var buttonWidth = floatingButton.offsetWidth;
      var buttonHeight = floatingButton.offsetHeight;

      currentX = Math.max(0, Math.min(currentX, windowWidth - buttonWidth));
      currentY = Math.max(0, Math.min(currentY, windowHeight - buttonHeight));

      floatingButton.style.left = currentX + "px";
      floatingButton.style.top = currentY + "px";
    }
  });

  document.addEventListener("mouseup", function () {
    isDragging = false;
  });

  // 点击悬浮球时执行转换并复制到剪贴板
  floatingButton.addEventListener("click", function () {
    var selector = getSelectorForCurrentWebsite();
    if (selector) {
      convertAndCopyMarkdown(selector);
    } else {
      var customSelector = prompt("请输入要选择的元素的CSS选择器：");
      if (customSelector) {
        convertAndCopyMarkdown(customSelector);
        saveCustomSelector(customSelector);
      } else {
        alert("未提供有效的选择器！");
      }
    }
  });

  // 根据当前网站获取相应的选择器
  function getSelectorForCurrentWebsite() {
    var currentURL = window.location.href;
    var selector = "";

    // 根据不同的网站URL设置对应的选择器
    if (currentURL.includes("example.com")) {
      selector = ".example-selector";
    } else if (currentURL.includes("anotherwebsite.com")) {
      selector = ".another-selector";
    } else {
      // 如果不是支持的网站，则尝试获取保存的自定义选择器
      selector = getSavedCustomSelector();
    }

    return selector;
  }

  // 将HTML转换为Markdown并复制到剪贴板
  function convertAndCopyMarkdown(selector) {
    var htmlElements = document.querySelectorAll(selector);
    if (htmlElements.length > 0) {
      var markdown = "";
      htmlElements.forEach(function (htmlElement) {
        var html = htmlElement.innerHTML;
        var convertedMarkdown = convertToMarkdown(html);
        markdown += convertedMarkdown + "\n\n";
      });
      copyToClipboard(markdown);
      alert("Markdown 已复制到剪贴板！");
    } else {
      alert("未找到匹配元素！");
    }
  }

  // 将HTML转换为Markdown
  function convertToMarkdown(html) {
    var turndownService = createTurndownServiceWithGfmSupport();
    var markdown = turndownService.turndown(html);
    return markdown;
  }

  // 创建带有GFM支持的TurndownService
  function createTurndownServiceWithGfmSupport() {
    var turndownService = new TurndownService();
    var gfm = turndownPluginGfm.gfm;

    turndownService.use(gfm);
    turndownService.addRule("codeBlock", {
      filter: function (node) {
        return (
          node.nodeName === "PRE" &&
          node.firstChild &&
          node.firstChild.nodeName === "CODE"
        );
      },
      replacement: function (content, node) {
        var className = node.firstChild.className || "";
        return "\n```" + className + "\n" + content + "\n```";
      },
    });

    return turndownService;
  }

  // 复制文本到剪贴板
  function copyToClipboard(text) {
    GM_setClipboard(text);
  }

  // 保存自定义选择器到脚本中
  function saveCustomSelector(selector) {
    localStorage.setItem("customSelector", selector);
  }

  // 获取保存的自定义选择器
  function getSavedCustomSelector() {
    return localStorage.getItem("customSelector") || "";
  }

  // 添加样式
  GM_addStyle(`
        div.floating-button {
            position: fixed;
            top: 50%;
            left: 0;
            z-index: 9999;
            background-color: #333;
            color: #fff;
            padding: 10px;
            cursor: pointer;
            transform: translateY(-50%);
        }
    `);
})();
